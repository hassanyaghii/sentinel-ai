STEP 1: DATABASE SETUP (ON 10.1.244.70)
--------------------------------------
Login to MySQL and initialize the required tables:

CREATE DATABASE IF NOT EXISTS sentinel_audit;
USE sentinel_audit;

CREATE TABLE audit_reports (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ip_address VARCHAR(45) NOT NULL,
    hostname VARCHAR(255),
    overall_score INT,
    summary TEXT,
    device_firmware VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE security_findings (
    id INT AUTO_INCREMENT PRIMARY KEY,
    report_id INT,
    risk_level VARCHAR(20),
    category VARCHAR(100),
    title VARCHAR(255),
    description TEXT,
    recommendation TEXT,
    FOREIGN KEY (report_id) REFERENCES audit_reports(id) ON DELETE CASCADE
);

STEP 2: LAUNCH BACKEND
----------------------
